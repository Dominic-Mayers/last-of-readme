// installer-driven adapter (general)
function getInstallationFingerprint({ configField, commandPrefix }) {
  const field = getPackageJsonField(configField, { allowEmpty: true });
  const hasConfigField = field !== null && field !== undefined && ...;
  const hooks = getCurrentScriptsHooks();
  const hooksWithInstallation = Object.entries(hooks)
    .filter(([, value]) => value.includes(commandPrefix))
    .map(([key]) => key);
  if (!hasConfigField && hooksWithInstallation.length === 0) return null;
  return { hasConfigField, hooksWithInstallation };
}

// LOR-driven adapter (application layer, both driven and driving)
function getLastOfReadmeInstallationFingerprint() {
  return getInstallationFingerprint({
    configField: 'lastOfReadme',
    commandPrefix: 'last-of-readme',
  });
}
